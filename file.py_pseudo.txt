

DEFINE FUNCTION room_calculation():

    SET (standard_room, deluxe_room, suite) TO (150, 180, 220)

    SET option TO INPUT("Would you like to book a room? (Yes/No) ").lower()

    IF option EQUALS "yes":

        OUTPUT("\nWhat type of room do you want?\nWe have the following options:\n1 - Standard Room :  Cost $150 per night\n2 - Deluxe Room :  Cost $180 per night\n3 - Suite :  Cost $220 per night\nPlease select 1, 2, or 3")

        SET x TO int(INPUT("Enter from the previous options: "))

        SET cost TO 0

        IF x EQUALS 1:

            SET cost TO standard_room

        ELSEIF x EQUALS 2:

            SET cost TO deluxe_room

        ELSEIF x EQUALS 3:

            SET cost TO suite

        ELSE:

            OUTPUT("Invalid Input! Please try again\n")

            CALL room_calculation() 

        SET stay_length TO INPUT("Enter length of stay (In days) : ")

        SET total_cost TO int(cost) * int(stay_length)

        RETURN total_cost

    ELSEIF option EQUALS "no":

        RETURN False

    ELSE:

        OUTPUT("Invalid Input! Please try again\n")

        CALL room_calculation() 



DEFINE FUNCTION rental_cars():

    OUTPUT("\nRental cars are $100 up to 24 hours then $60 FOR each additional day")

    SET x TO INPUT("Would you like to rent a car? (Yes/No) ").lower()

    SET rent_car_cost TO 0

    IF x EQUALS "yes":

        SET rent_days TO int(INPUT("\nEnter amount of days you will rent a car: "))

        IF rent_days >= 1:

            SET rent_car_cost TO 100 + ((rent_days - 1) * 60)

        ELSE:

            OUTPUT("Invalid Input! Please try again\n")

            CALL rental_cars() 

        RETURN rent_car_cost

    ELSEIF x EQUALS "no":

        SET rent_car_cost TO 0

    ELSE:

        OUTPUT("Invalid Input! Please try again\n")

        CALL rental_cars() 

    RETURN rent_car_cost



DEFINE FUNCTION theme_park_ticket():

    OUTPUT("\nTheme park parking prices are as follows:\n$175 FOR 1 day and $50 additional FOR each extended day")

    SET x TO INPUT("Would you like to buy parking? (Yes/No) ").lower()

    IF x EQUALS "yes":

        SET ticket_length TO int(INPUT("Enter duration of parking (In days): "))

        SET ticket_cost TO 0

        IF ticket_length >= 1:

            SET ticket_cost TO 175 + ((ticket_length - 1) * 50)

        ELSE:

            OUTPUT("Invalid Input! Please try again\n")

            CALL theme_park_ticket() 

        RETURN ticket_cost

    ELSEIF x EQUALS "no":

        RETURN False

    ELSE:

        OUTPUT("Invalid Input! Please try again\n")

        CALL theme_park_ticket() 



DEFINE FUNCTION theme_park_trip():

    SET x TO int(room_calculation())

    SET y TO int(rental_cars())

    SET z TO int(theme_park_ticket())

    OUTPUT("\nRoom Cost: ", x)

    OUTPUT("Rental Car Cost: ", y)

    OUTPUT("Theme Park Parking Cost: ", z)

    OUTPUT("Total theme park trip cost will be: ", (x + y + z))



CALL theme_park_trip()
